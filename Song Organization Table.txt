Song Organization
NOTE: Encrypted audio must be uncompressed!

	<Song featured=‘true/false'>
		<name>NAME</name>
		<album>ALBUM</album>
		<artist>ARTIST</artist>
		<genre>GENRE</genre>
		<format>FORMAT</format>
        	<downloadURL>URL</downloadURL>
		<link name=‘NAME’>URL</link>
		<releaseDate>DATE</releaseDate>
        	<smallArtworkURL>URL</smallArtworkURL>
        	<largeArtworkURL>URL</largeArtworkURL>
	</Song>


Example:

<Song featured='false'>
		<name>Liberty</name>
		<album>Silent Room</album>
		<artist>DJ Purify</artist>
		<genre>Trance</genre>
		<format>ogg</format>
        <downloadURL>https://copy.com/wGPdMB44oQSV</downloadURL>
		<link name='Bandcamp'>http://purify1.bandcamp.com/track/purify-liberty</link>
		<releaseDate>January 2, 2013</releaseDate>
        <smallArtworkURL>https://copy.com/VNkgncQtJ1QF</smallArtworkURL>
        <largeArtworkURL>https://copy.com/IirTDqQBtcaq</largeArtworkURL>
	</Song>
